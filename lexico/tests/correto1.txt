sumBelow10 :: [Integer] -> Integer;
sumBelow10 [] = 0;
sumBelow10 xs = do {
    sum <- 0;
    while(xs /= []) {
        x:xs <- xs;
        aux  <- if (x < 10) then {
                    yield x;
                } else {
                    yield 0;
                };
        sum <- sum + aux;
    };
    sum;
}
