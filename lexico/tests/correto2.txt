incListBy :: Integer -> [Integer] -> [Integer];
incListBy 0 xs = xs;
incListBy _ ([]) = []; 
incListBy i xs = do {
    ys <- [];
    while( xs /= [] ) {
        _:xs <- xs;
        ys <- (x + i):ys;
    }
    yield rev ys;
} where {
    rev ([]) = [];
    rev (x:xs) = (rev xs) ++ [x];
}
